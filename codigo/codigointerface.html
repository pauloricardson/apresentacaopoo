// Interfaces segregadas por responsabilidade

public interface INotificacaoEstoque {
    void notificarVacinaProximaValidade(LoteVacina lote);
    void notificarEstoqueBaixo(Vacina vacina);
}

public interface INotificacaoAgendamento {
    void notificarAgendamentoConfirmado(Agendamento agendamento);
    void notificarLembreteAgendamento(Agendamento agendamento);
}

public interface INotificacaoMassa {
    void notificarAgendamentosEmMassa(List agendamentos);
    void notificarCampanhaIniciada(CampanhaVacinacao campanha);
}

public interface IRelatorioAplicacoes {
    String gerarRelatorioTotalAplicadas();
    String gerarRelatorioEficaciaVacinal();
}

public interface IRelatorioCobertura {
    String gerarRelatorioCidadãosNaoVacinados();
    String gerarRelatorioCoberturaPorRegiao();
}

public interface IRelatorioEstoque {
    String gerarRelatorioConsumoPorUnidade();
    String gerarRelatorioLotesVencidos();
}
